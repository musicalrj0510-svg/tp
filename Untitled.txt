import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: Calculator()));

class Calculator extends StatefulWidget {
  @override
  _CalculatorState createState() => _CalculatorState();
}

class _CalculatorState extends State<Calculator> {
  double n1 = 0, n2 = 0;
  String res = "0";

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Simple Calculator"), backgroundColor: Colors.red),
      body: Padding(
        padding: EdgeInsets.all(20),
        child: Column(children: [
          TextField(onChanged: (v) => n1 = double.tryParse(v) ?? 0, decoration: InputDecoration(labelText: "Number 1")),
          TextField(onChanged: (v) => n2 = double.tryParse(v) ?? 0, decoration: InputDecoration(labelText: "Number 2")),
          Text("\nResult: $res\n", style: TextStyle(fontSize: 25)),
          Row(mainAxisAlignment: MainAxisAlignment.center, children: ["+", "-", "*", "/"].map((o) => 
            TextButton(child: Text(o), onPressed: () => setState(() {
              if (o == "+") res = "${n1 + n2}";
              if (o == "-") res = "${n1 - n2}";
              if (o == "*") res = "${n1 * n2}";
              if (o == "/") res = n2 != 0 ? "${n1 / n2}" : "Error";
            }))).toList())
        ]),
      ),
    );
  }
}

  String res = "";

  @override

    return Scaffold(
      appBar: AppBar(title: Text("Leap Year Checker"), backgroundColor: Colors.deepPurple),

import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: UserInfo()));

class UserInfo extends StatefulWidget {
  @override
  _UserInfoState createState() => _UserInfoState();
}

class _UserInfoState extends State<UserInfo> {
  String n = "", id = "", g = "Male";
  Map h = {"Playing": true, "Gaming": true, "Dancing": false};
  bool sub = false;

  @override
  Widget build(BuildContext context) => Scaffold(
    appBar: AppBar(title: Text("User Info"), backgroundColor: sub ? Colors.orange : Colors.red),
    body: sub ? Center(child: Text("Name: $n\nID: $id\nGender: $g\nHobbies: ${h.keys.where((k) => h[k])}", style: TextStyle(fontSize: 20, letterSpacing: 2))) 
              : ListView(padding: EdgeInsets.all(9), children: [
        for (var i in ["name", "ID"]) TextField(onChanged: (v) => i == "name" ? n = v : id = v, decoration: InputDecoration(labelText: "Enter $i", border: OutlineInputBorder(borderRadius: BorderRadius.circular(30)))),
        for (var v in ["Male", "Female"]) RadioListTile(title: Text(v), value: v, groupValue: g, onChanged: (x) => setState(() => g = x.toString())),
        ...h.keys.map((k) => CheckboxListTile(title: Text(k), value: h[k], onChanged: (v) => setState(() => h[k] = v))),
        ElevatedButton(onPressed: () => setState(() => sub = true), child: Text("Submit"))
      ]),
  );
}

import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: UserForm()));

class UserForm extends StatefulWidget {
  @override
  _UserFormState createState() => _UserFormState();
}

class _UserFormState extends State<UserForm> {
  final Map<String, String> data = {};
  final fields = ["Name", "ID", "Semester", "Dept", "City"];
  bool sub = false;

  @override
  Widget build(BuildContext context) => Scaffold(
    appBar: AppBar(title: Text("User Info"), backgroundColor: Colors.orange),
    body: Padding(
      padding: EdgeInsets.all(15),
      child: sub ? Column(children: fields.map((f) => Text("${f}: ${data[f]}", style: TextStyle(fontSize: 22, letterSpacing: 2))).toList())
                 : ListView(children: [
                    for (var f in fields) TextField(
                      onChanged: (v) => data[f] = v,
                      decoration: InputDecoration(labelText: "Enter $f", border: OutlineInputBorder(borderRadius: BorderRadius.circular(30))),
                    ),
                    ElevatedButton(onPressed: () => setState(() => sub = true), child: Text("Submit"))
                  ]),
    ),
  );
}
